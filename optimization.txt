                from datetime import datetime, timedelta
                from django.db import connection
                from celery import shared_task

                @shared_task
                def create_monthly_order_partition():
                    # Keyingi oy
                    today = datetime.today()
                    next_month = (today.replace(day=1) + timedelta(days=32)).replace(day=1)
                    table_name = f"orders_{next_month.year}_{next_month.month:02d}"

                    # Keyingi oy oxiri (exclusive)
                    following_month = (next_month + timedelta(days=32)).replace(day=1)

                    sql = f"""
                    CREATE TABLE IF NOT EXISTS {table_name} PARTITION OF orders
                    FOR VALUES FROM ('{next_month.date()}') TO ('{following_month.date()}');
                    """
                    with connection.cursor() as cursor:
                        cursor.execute(sql)
